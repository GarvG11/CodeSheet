  5. WAP to print the sum of 2 numbers 
                  
                 
import java.util.Scanner;  
class test
{
      public static void main(String args[])
      {
         Scanner sc = new Scanner(System.in);
         int a,b,c;
         System.out.print("Enter First Number");
         a = sc.nextInt();
         System.out.println("Enter second number");
         b = sc.nextInt();
         c = a + b;
         System.out.println("Sum of both the numbers is " + c);
      }
}

  7. WAP to print the the fibonacci series till N-th user input 
                  
import java.util.*;
class fibonacciSeries{
   public static void main(String args[]){ 
      Sanner sc = new Scanner(System.in);
      int n = sc.nextInt(); //user input
      int a, b, c, w; // declaring all the other variables 
      // giving values to all the variables 
      a = 0;
      b = 2;
      w = 0;
      // the while loop which will print the series
      while(w < n){

         System.out.print(a + " "); // printing the values
         c = a + b;
         a = b;
         b = c;
         w += 1;
      }
   }
}
